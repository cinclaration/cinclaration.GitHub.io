/*-------------- JavaScript/Section 2: Canvas Initialization ---------------*/ const canvas = document.getElementById("gameCanvas"); const ctx = canvas.getContext("2d"); function resizeCanvas() { canvas.width = window.innerWidth; canvas.height = window.innerHeight; document.querySelector('.background-layer').style.width = `${canvas.width}px`; document.querySelector('.background-layer').style.height = `${canvas.height}px`; } window.addEventListener('resize', resizeCanvas); resizeCanvas(); // Initial call /*-------------- JavaScript/Section 3: Background Setup ---------------*/ const background = document.querySelector('.background-layer'); let backgroundOffsetX = 0; let backgroundOffsetY = 0; function updateBackground() { backgroundOffsetX += 1; backgroundOffsetY += 1; background.style.backgroundPosition = `${-backgroundOffsetX}px ${-backgroundOffsetY}px`; } /*-------------- JavaScript/Section 4: UI Updates (Score and Timer) ---------------*/ let timeLeft = 30; let score = 0; function updateUI() { document.getElementById('score').textContent = score; document.getElementById('timer').textContent = `${timeLeft} seconds`; } function startTimer() { setInterval(() => { if (timeLeft > 0) { timeLeft--; updateUI(); } else { alert('Game Over! Try Again.'); resetGame(); } }, 1000); } function resetGame() { timeLeft = 30; score = 0; updateUI(); } /*-------------- JavaScript/Section 5: Spaceship and Controls ---------------*/ const spaceship = { x: canvas.width / 2, y: canvas.height / 2, radius: 20, dx: 0, dy: 0, friction: 0.9, speed: 0.15, img: new Image(), }; spaceship.img.src = 'https://i.imgur.com/x87KCsX.png'; function drawSpaceship() { if (spaceship.img.complete) { ctx.drawImage( spaceship.img, spaceship.x - spaceship.radius, spaceship.y - spaceship.radius, spaceship.radius * 2, spaceship.radius * 2 ); } else { ctx.beginPath(); ctx.arc(spaceship.x, spaceship.y, spaceship.radius, 0, Math.PI * 2); ctx.fillStyle = '#3498DB'; ctx.fill(); ctx.closePath(); } } function updateSpaceship() { spaceship.dx *= spaceship.friction; spaceship.dy *= spaceship.friction; spaceship.x += spaceship.dx; spaceship.y += spaceship.dy; spaceship.x = Math.max(spaceship.radius, Math.min(canvas.width - spaceship.radius, spaceship.x)); spaceship.y = Math.max(spaceship.radius, Math.min(canvas.height - spaceship.radius, spaceship.y)); drawSpaceship(); } window.addEventListener('deviceorientation', (event) => { const { beta, gamma } = event; spaceship.dx += gamma * spaceship.speed; spaceship.dy += beta * spaceship.speed; }); /*-------------- JavaScript/Section 6: Goals and Collision Detection ---------------*/ const goals = []; const goalImg = new Image(); goalImg.src = 'https://i.imgur.com/Xqmv2mE.jpeg'; function createGoals() { for (let i = 0; i < 5; i++) { const x = Math.random() * (canvas.width - 30) + 15; const y = Math.random() * (canvas.height - 30) + 15; goals.push({ x, y }); } } function drawGoals() { goals.forEach(goal => { ctx.drawImage(goalImg, goal.x - 15, goal.y - 15, 30, 30); }); } function checkCollisions() { goals.forEach((goal, index) => { if (Math.hypot(spaceship.x - goal.x, spaceship.y - goal.y) < spaceship.radius + 15) { score += 10; goals.splice(index, 1); } }); } /*-------------- JavaScript/Section 7: Game Loop ---------------*/ function gameLoop() { ctx.clearRect(0, 0, canvas.width, canvas.height); updateBackground(); updateSpaceship(); drawGoals(); checkCollisions(); updateUI(); requestAnimationFrame(gameLoop); } createGoals(); startTimer(); gameLoop();
